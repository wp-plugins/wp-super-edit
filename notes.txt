## Notes for adding specific admin head based on plugins from wp-hackers ##

READ: http://noteslog.com/post/how-to-load-javascript-in-wordpress-plugins/ 

all of the functions that add plugin menu items to the wp menu:

add_management_page(),
add_options_page(),
add_theme_page(),
add_users_page()

return a $hookname that you append to the action string:
"admin_print_scripts-"


function myplugin_menu() {
 if ( function_exists('add_management_page') ) {
   $page = add_management_page( 'myplugin', 'myplugin', 9, __FILE__, 'myplugin_admin_page' );
   add_action( "admin_print_scripts-$page", 'myplugin_admin_head' );
 }
}
